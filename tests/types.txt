double sqrt(double x)
{
	double guess = 1.0;
	double tmp;
	while (1) {
		tmp = guess * guess - x;
		tmp = (tmp < 0) ? 0 - tmp : tmp;
		if (tmp < 0.001)
			return guess;
		guess = (guess + x/guess) / 2;
	}
}
